```python
async def prioritize_gaps(score_result: dict, store_data: dict) -> list[dict]:
    """Identify gaps with explicit confidence."""
    gaps = []
    
    # For each dimension, find top gap
    for dim_name, dim_data in score_result.get("dimensions", {}).items():
        if dim_data.get("score", 0) < 50:  # Only flag if failing
            issues = dim_data.get("issues", [])
            if issues:
                gap = {
                    "dimension": dim_name,
                    "gap": issues[0],  # Top issue
                    "score": dim_data.get("score"),
                    "impact_score": _calc_impact(dim_name, dim_data.get("score")),
                    "effort_score": _estimate_effort(issues[0]),
                    "revenue_risk": _assess_revenue_risk(dim_name),
                    "priority_rank": 0,  # Will be set after sorting
                    "fix_suggestion": dim_data.get("recommendations", [""])[0],
                    "confidence": "high" if dim_data.get("score") < 30 else "medium"
                }
                gaps.append(gap)
    
    # VALIDATION: Only include gaps you can defend
    validated_gaps = []
    for gap in gaps:
        if gap["impact_score"] >= 5 and gap["effort_score"] <= 8:
            validated_gaps.append(gap)
    
    # Sort by: impact ÷ effort (highest first)
    for i, gap in enumerate(sorted(validated_gaps, 
                                   key=lambda g: g["impact_score"] / max(g["effort_score"], 1),
                                   reverse=True)):
        gap["priority_rank"] = i + 1
    
    logger.info(f"[GAP_ENGINE] Identified {len(validated_gaps)} defensible gaps")
    
    return validated_gaps[:7]  # Max 7 gaps (not 10)
```

```python
async def detect_temporal_drift(store_data: dict) -> dict:
    """Detect only HIGH-CONFIDENCE stale signals."""
    stale_signals = []
    
    # RULE 1: Only flag years more than 2 years old
    current_year = datetime.now().year
    for year in range(2000, current_year - 2):
        if f"{year}" in combined_text:
            # Check if it's actually a claim, not just a reference
            if any(phrase in combined_text for phrase in [
                f"new for {year}",
                f"launched {year}",
                f"founded {year}",
                f"since {year}"
            ]):
                stale_signals.append({
                    "text": f"Reference to year {year}",
                    "reason": "Outdated year reference — AI agents may think this is old data",
                    "risk": "Medium",
                    "type": "year_reference"
                })
    
    # RULE 2: Only flag policy contradictions if >3 day difference
    policy_days = re.findall(r"(\d+)\s+days?", policies_text)
    if policy_days and len(set(policy_days)) > 1:
        days_set = set(int(d) for d in policy_days)
        if max(days_set) - min(days_set) > 3:
            stale_signals.append({
                "text": f"Policy timeframes vary: {sorted(days_set)}",
                "reason": "Contradictory return windows — confuses AI agents about actual policy",
                "risk": "High",
                "type": "policy_contradiction"
            })
    
    logger.info(f"[TEMPORAL] Found {len(stale_signals)} HIGH-CONFIDENCE stale signals")
    
    return {
        "staleness_score": min(len(stale_signals) * 15, 100),
        "stale_signals": stale_signals,
        "summary": f"Detected {len(stale_signals)} temporal inconsistencies"
```

### Add Dual-Mode Analysis

```python
# In backend/app/routers/analyze.py, add this at the top:

@router.post("/store")
async def analyze_store(request: dict):
    """
    Analyze a Shopify store.
    
    Mode 1 (Public, no auth):
    { "store_url": "https://myfrido.com" }
    
    Mode 2 (Merchant with admin token):
    { 
        "store_domain": "myfrido.myshopify.com",
        "admin_token": "shpat_...",
        "mode": "admin"
    }
    """
    
    mode = request.get("mode", "public")
    
    if mode == "public":
        # Existing scraper logic
        store_url = request.get("store_url", "").strip()
        storefront_data = await scrape_storefront(store_url)
        store_data = await extract_store_data_from_scrape(storefront_data)
    
    elif mode == "admin":
        # NEW: Use Shopify Admin GraphQL API
        store_domain = request.get("store_domain", "").strip()
        admin_token = request.get("admin_token", "").strip()
        
        if not store_domain or not admin_token:
            raise HTTPException(status_code=400, detail="store_domain and admin_token required for admin mode")
        
        # Validate token (check it's not hardcoded placeholder)
        if admin_token.startswith("shpat_") and len(admin_token) > 100:
            client = ShopifyClient(store_domain, admin_token)
            store_data = await client.fetch_all()
            logger.info(f"[ANALYZE] Admin mode: fetched {len(store_data.get('products', []))} products from Shopify API")
        else:
            raise HTTPException(status_code=400, detail="Invalid admin token format")
    
    else:
        raise HTTPException(status_code=400, detail="mode must be 'public' or 'admin'")
    
    # Rest of analysis pipeline (same for both modes)
    scorer = StoreScorer(store_data)
    score_result = await scorer.score()
    gaps = await prioritize_gaps(score_result, store_data)
    temporal = await detect_temporal_drift(store_data)
    personas = await simulate_personas(store_data)
    
    return {
        "store": store_data.get("shop_info", {}).get("name", "Unknown"),
        "analysis_mode": mode,
        "score": score_result,
        "gaps": gaps,
        "temporal": temporal,
        "personas": personas,
        "status": "success"
    }
```
